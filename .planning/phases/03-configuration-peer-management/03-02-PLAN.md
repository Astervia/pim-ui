    ## Part A — `src/hooks/use-add-peer.ts`

    Create a lightweight hook that owns Sheet open state + form submission. Use `react-hook-form` for the form. The hook is a simple local-state container (useState for open + useState for in-flight); it does NOT touch the daemon-state atom.

    ```typescript
    import { useCallback, useState } from "react";
    import { useForm, type UseFormReturn } from "react-hook-form";
    import { toast } from "sonner";
    import { callDaemon } from "@/lib/rpc";
    import { RpcErrorCode, type RpcError } from "@/lib/rpc-types";
    // Checker Blocker 2 + D-30: refresh parsed TOML base after peer add so
    // raw-is-source-of-truth scans and Settings forms see the updated [[peers]].
    import { refetchSettingsConfig } from "@/hooks/use-settings-config";

    export interface AddPeerFormValues {
      address: string;
      mechanism: "tcp" | "bluetooth" | "wifi_direct";
      label: string;   // empty string → omit from RPC
    }

    interface UseAddPeerReturn {
      open: boolean;
      openSheet: () => void;
      closeSheet: () => void;
      form: UseFormReturn<AddPeerFormValues>;
      onSubmit: (values: AddPeerFormValues) => Promise<void>;
      submitting: boolean;
    }

    export function useAddPeer(): UseAddPeerReturn {
      const [open, setOpen] = useState(false);
      const [submitting, setSubmitting] = useState(false);
      const form = useForm<AddPeerFormValues>({
        defaultValues: { address: "", mechanism: "tcp", label: "" },
      });

      const openSheet = useCallback(() => {
        form.reset({ address: "", mechanism: "tcp", label: "" });
        setOpen(true);
      }, [form]);

      const closeSheet = useCallback(() => setOpen(false), []);

      const onSubmit = useCallback(async (values: AddPeerFormValues) => {
        setSubmitting(true);
        try {
          const params = {
            address: values.address.trim(),
            mechanism: values.mechanism,
            ...(values.label.trim() ? { label: values.label.trim() } : {}),
          };
          await callDaemon("peers.add_static", params);
          // D-30 / checker Blocker 2: re-fetch config.get so raw-is-source-of-truth
          // scans pick up the new [[peers]] entry before the Sheet closes.
          await refetchSettingsConfig();
          setOpen(false);
          form.reset();
        } catch (e) {
          const err = e as RpcError;
          if (err.code === RpcErrorCode.PeerAlreadyExists) {
            form.setError("address", { message: "That peer is already configured." });
          } else if (err.code === RpcErrorCode.InvalidPeerAddress) {
            form.setError("address", { message: "Address format not recognized by the daemon." });
          } else if (err.code === RpcErrorCode.InvalidParams) {
            toast.error("Daemon rejected parameters. [Show in Logs →]");
          } else {
            toast.error(err.message ?? "Couldn't add peer.");
          }
        } finally {
          setSubmitting(false);
        }
      }, [form]);

      return { open, openSheet, closeSheet, form, onSubmit, submitting };
    }
    ```

    ### Shared state between Sheet + ActionRow (state-pattern decision)

    `useAddPeer` must be shared between `AddPeerSheet` and `AddPeerActionRow` — both read/write the `open` flag + the submit handler + the react-hook-form instance. Use the **module-level atom + `useSyncExternalStore`** pattern established by Phase 2's `use-peer-detail.ts` (checker Info 2 — drop the earlier "contradictory useState scaffolding" block; one pattern only).

    Structure the atom as:
    - module-level `openState: boolean`, `submittingState: boolean`, `listeners: Set<() => void>`
    - module-level form holder: lazily initialize a single `useForm` instance via a module singleton. On first mount, call `form.reset(defaults)`; on close, call `form.reset()` to clear field errors but keep the same instance.
    - exported `useAddPeer()` returns the same shared instance for every caller.

    Do NOT use local `useState` for `open` / `submitting` / `form` — that produces independent instances per caller and breaks the Sheet/ActionRow sharing contract. Do NOT introduce a React context provider either (CONTEXT code_context §Established Patterns — atoms over context for global state in this codebase). The `useSyncExternalStore` + module atom is the single canonical pattern.

    ## Part A — `src/hooks/use-remove-peer.ts`

    Module-level atom for the selected peer (the row whose Remove button was clicked). Mirrors `use-add-peer.ts` shape. `peer_to_remove` is a `PeerSummary | null`. Open state is implicit: dialog is open iff `peer_to_remove !== null`.

    ```typescript
    import { useCallback, useSyncExternalStore } from "react";
    import { toast } from "sonner";
    import { callDaemon } from "@/lib/rpc";
    import { RpcErrorCode, type RpcError, type PeerSummary } from "@/lib/rpc-types";
    // Checker Blocker 2 + D-30: refresh parsed TOML base after peer remove so
    // raw-is-source-of-truth scans and Settings forms see the pruned [[peers]].
    import { refetchSettingsConfig } from "@/hooks/use-settings-config";

    // Module-level atom (mirrors use-daemon-state.ts pattern)
    let target: PeerSummary | null = null;
    let submitting = false;
    const listeners = new Set<() => void>();
    function notify() { listeners.forEach(fn => fn()); }
    function subscribe(cb: () => void) { listeners.add(cb); return () => { listeners.delete(cb); }; }
    function getTarget(): PeerSummary | null { return target; }
    function getSubmitting(): boolean { return submitting; }

    export interface UseRemovePeerReturn {
      peer: PeerSummary | null;
      submitting: boolean;
      requestRemove: (peer: PeerSummary) => void;
      cancel: () => void;
      confirm: () => Promise<void>;
    }

    export function useRemovePeer(): UseRemovePeerReturn {
      const peer = useSyncExternalStore(subscribe, getTarget, getTarget);
      const submittingNow = useSyncExternalStore(subscribe, getSubmitting, getSubmitting);

      const requestRemove = useCallback((p: PeerSummary) => { target = p; notify(); }, []);
      const cancel = useCallback(() => { target = null; notify(); }, []);
      const confirm = useCallback(async () => {
        if (!target) return;
        const toRemove = target;
        submitting = true; notify();
        try {
          // Prefer config_entry_id when available (the daemon returned it on add_static).
          // PeerSummary does not carry config_entry_id directly in Phase 1 types — fall
          // back to node_id. The daemon accepts either per PeersRemoveParams.
          await callDaemon("peers.remove", { node_id: toRemove.node_id });
          // D-30 / checker Blocker 2: re-fetch config.get so the Settings
          // scaffold sees the pruned [[peers]] array on next open.
          await refetchSettingsConfig();
          target = null;
        } catch (e) {
          const err = e as RpcError;
          if (err.code === RpcErrorCode.PeerNotFound) {
            toast.message("Peer was already removed.");
            target = null;  // race — close the dialog
          } else {
            toast.error(err.message ?? "Couldn't remove peer.");
          }
        } finally {
          submitting = false;
          notify();
        }
      }, []);

      return { peer, submitting: submittingNow, requestRemove, cancel, confirm };
    }
    ```

    NOTE on `config_entry_id` vs `node_id`: CONTEXT D-19 says to prefer `config_entry_id` when available, but the `PeerSummary` type in `rpc-types.ts` does not expose it. The daemon's `peers.remove` accepts either (`PeersRemoveParams` has both fields optional, one required). Using `node_id` is acceptable for Phase 3 — it's unambiguous for connected peers. If the live daemon returns `config_entry_id` in a later schema update, re-wire then.

    ```tsx
    import type { MouseEvent } from "react";
    import { Button } from "@/components/ui/button";
    import { useDaemonState } from "@/hooks/use-daemon-state";
    import { useRemovePeer } from "@/hooks/use-remove-peer";
    import type { PeerSummary } from "@/lib/rpc-types";

    export function PeerRemoveButton({ peer }: { peer: PeerSummary }) {
      const { requestRemove } = useRemovePeer();
      const { snapshot } = useDaemonState();
      const limited = snapshot.state !== "running";
      const onClick = (e: MouseEvent<HTMLButtonElement>) => {
        e.stopPropagation();
        if (!limited) requestRemove(peer);
      };
      return (
        <Button
          type="button"
          variant="ghost"
          size="sm"
          className="border border-border hover:border-destructive hover:text-destructive hover:bg-transparent"
          disabled={limited}
          onClick={onClick}
          title={limited ? "Reconnect to remove peers." : undefined}
        >
          [ Remove ]
        </Button>
      );
    }
    ```
