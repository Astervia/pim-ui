    ## Part A — `src/hooks/use-log-filters.ts`

    Module-level atom for search text + time range. 300ms debounce lives here (not in the consumer) via a simple setTimeout pattern — the atom exposes the "live" text for the input binding AND the debounced text that feeds the filter chain.

    ```typescript
    import { useCallback, useEffect, useSyncExternalStore } from "react";
    // Phase 2's useLogsStream exposes setLevel/setPeer; applyLogsFilter
    // must reach them. They already live on module-level atoms in Phase 2;
    // import the module-level setters (NOT the hook) to write them directly.
    // If Phase 2 did not export module-level setters, fall back to stashing
    // the preset in this atom and having useLogsStream consume it on mount.
    import { setLevelAtom, setPeerAtom, setSourceAtom } from "./use-logs-stream";

    export type TimeRangePreset = "last_5m" | "last_15m" | "last_1h" | "all" | "custom";

    export type LogTimeRange =
      | { kind: "preset"; preset: TimeRangePreset }
      | { kind: "custom"; from: string; to: string };

    export interface LogsFilterPreset {
      source?: string;
      level?: "trace" | "debug" | "info" | "warn" | "error";
      peer_id?: string | null;
      text?: string;
    }

    // Module-level atom
    let searchTextLive = "";
    let searchTextDebounced = "";
    let timeRange: LogTimeRange = { kind: "preset", preset: "all" };
    let debounceTimer: ReturnType<typeof setTimeout> | null = null;
    const listeners = new Set<() => void>();

    function notify() { listeners.forEach(fn => fn()); }
    function subscribe(cb: () => void) { listeners.add(cb); return () => { listeners.delete(cb); }; }
    function getSearchLive() { return searchTextLive; }
    function getSearchDebounced() { return searchTextDebounced; }
    function getTimeRange() { return timeRange; }

    /**
     * Non-hook module function for [Show in Logs →] toast actions
     * (checker Warning 5). Updates filter atoms but does NOT navigate —
     * callers pair this with setActive("logs") from use-active-screen.
     *
     * Implementation: write to this atom for search/text, delegate level /
     * peer / source to use-logs-stream's module atoms. If Phase 2 does not
     * export module-level setters, extend it — this is the ONLY canonical
     * routing path and the contract is stable.
     */
    export function applyLogsFilter(preset: LogsFilterPreset): void {
      if (preset.text !== undefined) {
        searchTextLive = preset.text;
        searchTextDebounced = preset.text;
      }
      if (preset.level !== undefined && typeof setLevelAtom === "function") {
        setLevelAtom(preset.level);
      }
      if (preset.peer_id !== undefined && typeof setPeerAtom === "function") {
        setPeerAtom(preset.peer_id);
      }
      if (preset.source !== undefined && typeof setSourceAtom === "function") {
        setSourceAtom(preset.source);
      }
      notify();
    }

    export interface UseLogFiltersReturn {
      /** Live value bound to the input — re-renders on every keystroke. */
      searchText: string;
      /** Debounced value fed to the filter chain. */
      searchTextDebounced: string;
      setSearchText: (s: string) => void;
      timeRange: LogTimeRange;
      setTimeRange: (r: LogTimeRange) => void;
    }

    export function useLogFilters(): UseLogFiltersReturn {
      const live = useSyncExternalStore(subscribe, getSearchLive, getSearchLive);
      const debounced = useSyncExternalStore(subscribe, getSearchDebounced, getSearchDebounced);
      const range = useSyncExternalStore(subscribe, getTimeRange, getTimeRange);

      const setSearchText = useCallback((s: string) => {
        searchTextLive = s;
        if (debounceTimer) clearTimeout(debounceTimer);
        debounceTimer = setTimeout(() => {
          searchTextDebounced = s;
          debounceTimer = null;
          notify();
        }, 300);
        notify();  // fire once immediately so the input value updates live
      }, []);

      const setTR = useCallback((r: LogTimeRange) => {
        timeRange = r;
        notify();
      }, []);

      useEffect(() => () => {
        // No-op cleanup — the atom lives for the app lifetime. Debounce
        // timer is cleared implicitly when the next setSearchText fires.
      }, []);

      return {
        searchText: live,
        searchTextDebounced: debounced,
        setSearchText,
        timeRange: range,
        setTimeRange: setTR,
      };
    }
    ```

    ## Part D — `src/components/logs/custom-time-range-dialog.tsx`

    Uses `dialog.tsx` (NOT alert-dialog — per 03-UI-SPEC §S7 it's a non-destructive filter choice). From/To = `<input type="time">` per D-22.

    ```tsx
    import { useEffect, useState } from "react";
    import { Dialog, DialogContent, DialogDescription, DialogFooter, DialogHeader, DialogTitle } from "@/components/ui/dialog";
    import { Button } from "@/components/ui/button";
    import { useLogFilters } from "@/hooks/use-log-filters";
    import { useLogsStream } from "@/hooks/use-logs-stream";

    export function CustomTimeRangeDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (v: boolean) => void }) {
      const { timeRange, setTimeRange } = useLogFilters();
      const { rows } = useLogsStream();    // to seed defaults from the buffer
      const [from, setFrom] = useState("");
      const [to, setTo] = useState("");
      // Remember the preset the user was on before opening Custom… so Cancel
      // can revert (checker Info 1). If they were already on a custom range,
      // revert to "all".
      const [previousPreset, setPreviousPreset] = useState<"last_5m" | "last_15m" | "last_1h" | "all">("all");

      useEffect(() => {
        if (!open) return;
        // Capture previous preset when opening so Cancel can revert.
        if (timeRange.kind === "preset" && timeRange.preset !== "custom") {
          setPreviousPreset(timeRange.preset as "last_5m" | "last_15m" | "last_1h" | "all");
        }
        if (rows.length === 0) {
          setFrom(""); setTo(""); return;
        }
        const oldest = new Date(rows[0].ts);
        const newest = new Date(rows[rows.length - 1].ts);
        const hhmm = (d: Date) => `${String(d.getHours()).padStart(2, "0")}:${String(d.getMinutes()).padStart(2, "0")}`;
        setFrom(hhmm(oldest)); setTo(hhmm(newest));
      }, [open, rows, timeRange]);

      const apply = () => {
        if (!from || !to) return;
        const now = new Date();
        const mkIso = (hhmm: string) => {
          const [h, m] = hhmm.split(":").map(Number);
          const d = new Date(now);
          d.setHours(h, m, 0, 0);
          return d.toISOString();
        };
        setTimeRange({ kind: "custom", from: mkIso(from), to: mkIso(to) });
        onOpenChange(false);
      };

      return (
        <Dialog open={open} onOpenChange={onOpenChange}>
          <DialogContent>
            <DialogHeader>
              <DialogTitle>Filter by time range</DialogTitle>
            </DialogHeader>
            <div className="flex flex-col gap-4">
              <label className="flex flex-col gap-2 font-mono text-sm">
                From
                <input type="time" value={from} onChange={(e) => setFrom(e.target.value)} className="bg-transparent border border-border text-foreground font-mono text-sm px-3 py-2" />
              </label>
              <label className="flex flex-col gap-2 font-mono text-sm">
                To
                <input type="time" value={to} onChange={(e) => setTo(e.target.value)} className="bg-transparent border border-border text-foreground font-mono text-sm px-3 py-2" />
              </label>
            </div>
            <DialogFooter>
              <Button type="button" variant="ghost" onClick={() => { setTimeRange({ kind: "preset", preset: previousPreset }); onOpenChange(false); }}>[ Cancel ]</Button>
              <Button type="button" variant="default" onClick={apply} disabled={!from || !to}>[ Apply ]</Button>
            </DialogFooter>
          </DialogContent>
        </Dialog>
      );
    }
    ```

    ## Part F — Extend `src/hooks/use-logs-stream.ts`

    Consume `useLogFilters` and apply the filter chain in this order:
    1. Level — already server-side (existing Phase 2 behavior)
    2. Peer — already client-side (existing Phase 2 behavior)
    3. **Text search** (new) — case-insensitive substring match against `row.message + " " + row.source + " " + (row.peer_id ?? "")`; uses `searchTextDebounced`

    ## Part A — `src/lib/debug-snapshot.ts`

    Builds the snapshot JSON and triggers the download. No DOM listener beyond the created `<a>` click.

    ```typescript
    /**
     * Debug snapshot builder (OBS-03 / D-23 / D-24).
     *
     * Schema: snake_case verbatim so pasting into a kernel-repo bug
     * report diffs cleanly against `pim status --json` +
     * `pim logs --json` output. Version-bumped if the schema changes.
     */
    import type { LogEvent, LogLevel, PeerDiscovered, PeerSummary, Status } from "./rpc-types";

    export interface DebugSnapshot {
      snapshot_version: 1;
      ui_version: string;
      captured_at: string;                  // ISO-8601 with milliseconds
      daemon_status: Status | null;
      peers: PeerSummary[];
      discovered: PeerDiscovered[];
      logs: LogEvent[];
      filters_applied: {
        level: LogLevel;
        peer_id: string | null;
        text: string;
        time_range: string;
      };
    }

    export interface BuildSnapshotInput {
      daemonStatus: Status | null;
      peers: PeerSummary[];
      discovered: PeerDiscovered[];
      logs: LogEvent[];
      filters: {
        level: LogLevel;
        peer_id: string | null;
        text: string;
        time_range: string;
      };
    }

    /** Build the DebugSnapshot object in memory. Pure — no DOM. */
    export function buildDebugSnapshot(input: BuildSnapshotInput): DebugSnapshot {
      return {
        snapshot_version: 1,
        ui_version: import.meta.env.VITE_APP_VERSION ?? "unknown",
        captured_at: new Date().toISOString(),
        daemon_status: input.daemonStatus,
        peers: input.peers,
        discovered: input.discovered,
        logs: input.logs,
        filters_applied: input.filters,
      };
    }

    /** Per D-24: colons replaced with hyphens so Windows can save the file. */
    export function snapshotFilename(capturedAt: string): string {
      return `pim-debug-snapshot-${capturedAt.replace(/[:]/g, "-")}.json`;
    }

    /**
     * Serialize + trigger download via Blob + <a download>. Revokes the
     * object URL on a microtask so the browser has time to start the
     * download (D-23).
     */
    export function downloadSnapshot(snapshot: DebugSnapshot): void {
      const json = JSON.stringify(snapshot, null, 2);
      const blob = new Blob([json], { type: "application/json" });
      const url = URL.createObjectURL(blob);
      const a = document.createElement("a");
      a.href = url;
      a.download = snapshotFilename(snapshot.captured_at);
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
      // Revoke on microtask — the browser has already begun the download
      // but the URL is still valid until the next tick.
      queueMicrotask(() => URL.revokeObjectURL(url));
    }
    ```

    import { useState } from "react";

    import { Button } from "@/components/ui/button";

    function timeRangeLabel(r: LogTimeRange): string {
      if (r.kind === "custom") return `Custom (${r.from} – ${r.to})`;
      switch (r.preset) {
        case "last_5m":  return "Last 5 min";
        case "last_15m": return "Last 15 min";
        case "last_1h":  return "Last 1 hour";
        case "all":      return "All session";
        case "custom":   return "Custom…";
      }
    }

    export function DebugSnapshotButton({ className }: { className?: string }) {
      const [preparing, setPreparing] = useState(false);
      const { snapshot } = useDaemonState();
      const { rows, level, peer } = useLogsStream();
      const { searchText, timeRange } = useLogFilters();
      const { setActive } = useActiveScreen();

      const onClick = () => {
        setPreparing(true);
        try {
          const built = buildDebugSnapshot({
            daemonStatus: snapshot.status,
            peers: snapshot.status?.peers ?? [],
            discovered: snapshot.discovered,
            logs: rows,
            filters: {
              level,
              peer_id: peer,
              text: searchText,
              time_range: timeRangeLabel(timeRange),
            },
          });
          downloadSnapshot(built);
          toast.success(`Snapshot saved as ${snapshotFilename(built.captured_at)}`, { duration: 4000 });
        } catch {
          toast.error("Couldn't generate snapshot.", {
            duration: 8000,
            description: "[Show in Logs →]",
            // Checker Warning 5: wire the [Show in Logs →] action directly.
            action: {
              label: "Show in Logs →",
              onClick: () => {
                setActive("logs");
                // No source filter here — snapshot failures are UI-side
                // errors, not daemon-source filterable events.
                applyLogsFilter({});
              },
            },
          });
        } finally {
          setPreparing(false);
        }
      };

      return (
        <Button type="button" variant="default" onClick={onClick} disabled={preparing} className={className}>
          {preparing ? "[ Preparing… ]" : "[ Export debug snapshot ]"}
        </Button>
      );
    }
