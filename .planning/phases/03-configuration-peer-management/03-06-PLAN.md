    ## Part A — `src/hooks/use-raw-toml-save.ts`

    Orchestrates dry_run-first save of the textarea buffer verbatim. On reject, stores `errors[]` as `RawTomlError[]` so the gutter + inline rows can render them. On success, refetches config via useSettingsConfig.

    ```typescript
    import { useCallback, useState } from "react";
    import { toast } from "sonner";
    import { callDaemon } from "@/lib/rpc";
    import { RpcErrorCode, type ConfigValidationError, type RpcError } from "@/lib/rpc-types";
    import { useSettingsConfig } from "./use-settings-config";
    import { useDaemonState } from "./use-daemon-state";
    import { useActiveScreen } from "./use-active-screen";
    import { applyLogsFilter } from "./use-log-filters";    // checker Warning 5
    import { restartDaemon } from "@/lib/daemon-restart";   // checker Warning 3 — shared util

    export interface RawTomlError {
      line: number;
      column: number;
      path: string;
      message: string;
    }

    export type RawSaveState = "idle" | "saving" | "saved" | "error";

    export function useRawTomlSave() {
      const [state, setState] = useState<RawSaveState>("idle");
      const [errors, setErrors] = useState<RawTomlError[]>([]);
      const { refetch } = useSettingsConfig();
      const { actions } = useDaemonState();
      const { setActive } = useActiveScreen();

      const save = useCallback(async (buffer: string) => {
        setState("saving"); setErrors([]);
        try {
          await callDaemon("config.save", { format: "toml", config: buffer, dry_run: true });
          const real = await callDaemon("config.save", { format: "toml", config: buffer, dry_run: false });
          await refetch();
          if (real.requires_restart.length > 0) {
            // Checker Warning 3: shared restartDaemon util — no more no-op on the
            // raw-TOML [ Restart ] toast action.
            toast(`Saved. Restart pim to apply: ${real.requires_restart.join(", ")}`, {
              duration: 8000,
              action: {
                label: "[ Restart ]",
                onClick: () => { void restartDaemon(actions); },
              },
            });
          } else {
            toast.success("Saved.", { duration: 3000 });
          }
          setState("saved");
          setTimeout(() => setState("idle"), 2000);
        } catch (e) {
          const err = e as RpcError;
          if (err.code === RpcErrorCode.ConfigValidationFailed || err.code === RpcErrorCode.ConfigSaveRejected) {
            const list = (err.data as ConfigValidationError[] | undefined) ?? [];
            setErrors(list.map(e => ({ line: e.line, column: e.column, path: e.path, message: e.message })));
            // Checker Warning 5: wire the [Show in Logs →] action handler.
            toast.error(`Daemon rejected settings: ${list[0]?.message ?? err.message}`, {
              duration: 8000,
              description: "[Show in Logs →]",
              action: {
                label: "Show in Logs →",
                onClick: () => {
                  setActive("logs");
                  applyLogsFilter({ source: "config" });
                },
              },
            });
          } else {
            toast.error(err.message ?? "Save failed.");
          }
          setState("error");
          setTimeout(() => setState("idle"), 2000);
        }
      }, [refetch, actions, setActive]);

      return { state, errors, save };
    }
    ```

    ## Part B — `src/components/settings/raw-toml-gutter.tsx`

    ```tsx
    import { cn } from "@/lib/utils";

    export interface RawTomlGutterProps {
      lineCount: number;
      errorLines: Set<number>;                            // 1-based line numbers that have errors
      onErrorClick: (line: number) => void;
    }

    export function RawTomlGutter({ lineCount, errorLines, onErrorClick }: RawTomlGutterProps) {
      return (
        <div
          className="w-12 bg-muted/30 border-r border-border text-right font-code text-xs text-muted-foreground select-none"
          aria-hidden="true"
        >
          {Array.from({ length: lineCount }, (_, i) => {
            const line = i + 1;
            const hasError = errorLines.has(line);
            return (
              <div
                key={line}
                className={cn("leading-[22px] px-1", hasError && "text-accent cursor-pointer")}
                onClick={hasError ? () => onErrorClick(line) : undefined}
                role={hasError ? "button" : undefined}
                aria-label={hasError ? `jump to error on line ${line}` : undefined}
              >
                {hasError ? `⚠ ${line}` : line}
              </div>
            );
          })}
        </div>
      );
    }
    ```

    ## Part C — `src/components/settings/raw-toml-editor.tsx`

    The complete editor surface per 03-UI-SPEC §S1b. Meta row + editor frame + textarea + gutter + inline error rows + `[ Save ]` footer.

    ```tsx
    import { useEffect, useRef, useState, useMemo } from "react";
    import { Button } from "@/components/ui/button";
    import { RawTomlGutter } from "./raw-toml-gutter";
    import { useSettingsConfig } from "@/hooks/use-settings-config";
    import { useRawTomlSave } from "@/hooks/use-raw-toml-save";
    import { parseToml } from "@/lib/config/parse-toml";

    function offsetOfLineCol(text: string, line: number, column: number): number {
      const lines = text.split("\n");
      let off = 0;
      for (let i = 0; i < Math.min(line - 1, lines.length); i++) off += lines[i].length + 1;
      return off + Math.max(0, column - 1);
    }

    export function RawTomlEditor() {
      const { raw, sourcePath, lastModified } = useSettingsConfig();
      const { state, errors, save } = useRawTomlSave();
      const [buffer, setBuffer] = useState(raw);
      const taRef = useRef<HTMLTextAreaElement | null>(null);

      // Sync buffer when a fresh `raw` arrives (after save refetch).
      useEffect(() => { setBuffer(raw); }, [raw]);

      const lineCount = useMemo(() => buffer.split("\n").length, [buffer]);
      const errorLines = useMemo(() => new Set(errors.map(e => e.line)), [errors]);
      const dirty = buffer !== raw;
      const parseCheck = useMemo(() => parseToml(buffer), [buffer]);
      const unparseable = !parseCheck.ok && errors.length === 0 && !dirty;   // only show if daemon-returned raw is itself unparseable

      const onErrorClick = (line: number) => {
        const ta = taRef.current;
        if (!ta) return;
        const err = errors.find(e => e.line === line);
        const offset = offsetOfLineCol(buffer, line, err?.column ?? 1);
        ta.focus();
        ta.setSelectionRange(offset, offset);
      };

      const saveLabel = state === "saving" ? "[ Saving… ]" : state === "saved" ? "[ Saved ]" : "[ Save ]";
      const saveDisabled = state === "saving" || state === "saved" || !dirty;

      return (
        <div>
          {unparseable && (
            <div className="mb-3 p-3 border border-border bg-popover font-mono text-sm">
              <p className="text-destructive">Couldn't parse TOML returned by daemon.</p>
            </div>
          )}

          <div className="flex items-center justify-between text-xs text-muted-foreground mb-3 font-mono">
            <span>source: {sourcePath}</span>
            <span>last modified {lastModified}</span>
          </div>

          <div className="border border-border bg-popover flex">
            <RawTomlGutter lineCount={lineCount} errorLines={errorLines} onErrorClick={onErrorClick} />
            <textarea
              ref={taRef}
              value={buffer}
              onChange={(e) => setBuffer(e.target.value)}
              spellCheck={false}
              wrap="off"
              aria-label="raw pim.toml configuration"
              className="font-code text-sm leading-[22px] bg-popover text-foreground p-3 resize-none w-full min-h-[400px] outline-none"
            />
          </div>

          {errors.length > 0 && (
    ```

    ```tsx
    function relative(iso: string): string {
      if (!iso) return "";
      const t = new Date(iso).getTime();
      const diff = Date.now() - t;
      const min = Math.floor(diff / 60_000);
      if (min < 1) return "just now";
      if (min < 60) return `${min}m ago`;
      const h = Math.floor(min / 60);
      if (h < 24) return `${h}h ago`;
      const d = Math.floor(h / 24);
      return `${d}d ago`;
    }

    export function AdvancedSection({ open, onOpenChange }: { open: boolean; onOpenChange: (v: boolean) => void }) {
      const { raw, lastModified } = useSettingsConfig();
      const lineCount = raw.split("\n").length;
      const summary = `${lineCount} lines · last saved ${relative(lastModified)}`;
      return (
        <CollapsibleCliPanel id="advanced" title="ADVANCED — RAW CONFIG" summary={summary} open={open} onOpenChange={onOpenChange}>
          <div id="settings-section-advanced">
            <RawTomlEditor />
          </div>
    ```
