    ## Part C — `src/components/settings/section-save-footer.tsx`

    Shared footer row: dirty-dot + `[ Save ]` button with state label + LimitedMode hint. Copy verbatim per 03-UI-SPEC §Primary CTAs table:

    ```tsx
    import { Button } from "@/components/ui/button";
    import { useDaemonState } from "@/hooks/use-daemon-state";
    import type { SaveState } from "@/hooks/use-section-save";

    export function SectionSaveFooter({
      dirty,
      state,
      onSave,
    }: {
      dirty: boolean;
      state: SaveState;
      onSave: () => void;
    }) {
      const { snapshot } = useDaemonState();
      const limited = snapshot.state !== "running";
      const label = state === "saving" ? "[ Saving… ]" : state === "saved" ? "[ Saved ]" : "[ Save ]";
      const disabled = limited || !dirty || state === "saving" || state === "saved";

      return (
        <div className="mt-6 flex flex-col gap-2">
          {limited && (
            <p className="font-mono text-sm text-muted-foreground">Daemon stopped — reconnect to save.</p>
          )}
          <div className="flex items-center justify-end gap-2">
            {dirty && !limited && <span aria-label="unsaved changes" role="img" className="text-primary font-mono">·</span>}
            <Button type="button" variant="default" disabled={disabled} aria-busy={state === "saving" || undefined} onClick={onSave}>
              {label}
            </Button>
          </div>
        </div>
      );
    }
    ```

    ## Part G — `src/hooks/use-pending-restart.ts`

    D-26: persist per-section pending-restart field sets to `localStorage` under key `pim-ui.pending-restart`; clear on `status.event { kind: "role_changed" }`.

    **Invariant (checker Info 3):** the module atom + stable-reference handler pattern means that regardless of how many section components mount, the W1 fan-out `actions.subscribe("status.event", handler)` call registers exactly ONE handler at app-init time (the first mount) and it lives for the app lifetime. N section mounts do NOT register N handlers — the module-level subscribe guard ensures single-registration. The executor MUST add a one-line comment in `use-pending-restart.ts` documenting this invariant so future edits don't accidentally break it.

    ```typescript
    import { useCallback, useEffect, useSyncExternalStore } from "react";
    import { type SectionId, SECTION_IDS } from "@/lib/config/section-schemas";
    import { useDaemonState } from "@/hooks/use-daemon-state";

    const LS_KEY = "pim-ui.pending-restart";
    type PendingMap = Record<SectionId, string[]>;

    function loadFromStorage(): PendingMap {
      const out = Object.fromEntries(SECTION_IDS.map(id => [id, [] as string[]])) as PendingMap;
      try {
        const raw = localStorage.getItem(LS_KEY);
        if (!raw) return out;
        const parsed = JSON.parse(raw) as Partial<PendingMap>;
        for (const id of SECTION_IDS) {
          out[id] = Array.isArray(parsed[id]) ? (parsed[id] as string[]) : [];
        }
      } catch { /* noop */ }
      return out;
    }

    let atom: PendingMap = typeof localStorage !== "undefined" ? loadFromStorage() : (Object.fromEntries(SECTION_IDS.map(id => [id, [] as string[]])) as PendingMap);
    const listeners = new Set<() => void>();
    function notify() { listeners.forEach(fn => fn()); }
    function persist() { localStorage.setItem(LS_KEY, JSON.stringify(atom)); }
    function subscribe(cb: () => void) { listeners.add(cb); return () => { listeners.delete(cb); }; }
    function get() { return atom; }

    // INVARIANT (checker Info 3): single status.event subscription for the
    // lifetime of the app. Module-level guard ensures N section mounts register
    // exactly 1 handler via stable reference — NOT N handlers. The subscription
    // never unsubscribes (it's tied to the app process, not a component tree).
    let statusEventSubscribed = false;
    function ensureStatusEventSubscription(actions: ReturnType<typeof useDaemonState>["actions"]): void {
      if (statusEventSubscribed) return;
      statusEventSubscribed = true;
      void actions.subscribe("status.event", (evt) => {
        if (evt.kind === "role_changed") {
          atom = Object.fromEntries(SECTION_IDS.map(id => [id, [] as string[]])) as PendingMap;
          persist(); notify();
        }
      });
    }

    export function usePendingRestart(sectionId: SectionId) {
      const map = useSyncExternalStore(subscribe, get, get);
      const { actions } = useDaemonState();

      // Register the fan-out handler exactly once — subsequent mounts are no-ops.
      useEffect(() => {
        ensureStatusEventSubscription(actions);
      }, [actions]);

      const addFields = useCallback((sid: SectionId, fields: string[]) => {
        const existing = atom[sid] ?? [];
        atom = { ...atom, [sid]: Array.from(new Set([...existing, ...fields])) };
        persist(); notify();
      }, []);
      const clearAll = useCallback(() => {
        atom = Object.fromEntries(SECTION_IDS.map(id => [id, [] as string[]])) as PendingMap;
        persist(); notify();
      }, []);

      return { fields: map[sectionId], addFields, clearAll };
    }
    ```

    Note: `actions.subscribe` goes through `useDaemonState`'s fan-out map — this is NOT a new `listen(...)` call. Grep for `listen(` in this file should return 0.

    ## Part H.1 — `src/hooks/use-dirty-sections.ts` (addresses checker Blocker 1)

    Module atom that tracks per-section dirty state + dirty-field-count. `useSectionSave` writes into it via a useEffect on `form.formState.isDirty`; `active-screen.tsx` reads it before every navigation; `stop-confirm-dialog.tsx` reads it before opening the Stop dialog.

    ```typescript
    import { useSyncExternalStore } from "react";
    import { type SectionId, SECTION_IDS } from "@/lib/config/section-schemas";

    interface DirtyEntry { dirty: boolean; dirtyFieldCount: number; }
    type DirtyMap = Record<SectionId, DirtyEntry>;

    let atom: DirtyMap = Object.fromEntries(
      SECTION_IDS.map(id => [id, { dirty: false, dirtyFieldCount: 0 }]),
    ) as DirtyMap;
    const listeners = new Set<() => void>();
    function notify() { listeners.forEach(fn => fn()); }
    function subscribe(cb: () => void) { listeners.add(cb); return () => { listeners.delete(cb); }; }
    function get() { return atom; }

    /**
     * Non-hook writer — called by use-section-save's useEffect on every
     * form.formState change. Module-scope keeps writes outside render.
     */
    export function setSectionDirty(id: SectionId, dirty: boolean, dirtyFieldCount: number): void {
      const prev = atom[id];
      if (prev.dirty === dirty && prev.dirtyFieldCount === dirtyFieldCount) return;
      atom = { ...atom, [id]: { dirty, dirtyFieldCount } };
      notify();
    }

    /** Returns the list of sections currently dirty. */
    export function getDirtySections(): Array<{ id: SectionId; dirtyFieldCount: number }> {
      return SECTION_IDS
        .filter(id => atom[id].dirty)
        .map(id => ({ id, dirtyFieldCount: atom[id].dirtyFieldCount }));
    }

    /**
     * Fan-out reset — active-screen / stop-confirm-dialog calls this to
     * notify every mounted section form to call form.reset() before
     * navigation proceeds. Sections listen via a window CustomEvent
     * `pim:settings-discard-reset` with detail.id === sectionId (or "all").
     */
    export function emitDiscardReset(id: SectionId | "all"): void {
      window.dispatchEvent(new CustomEvent("pim:settings-discard-reset", { detail: { id } }));
      // Optimistically clear the atom — the form.reset() handlers will
      // echo this via setSectionDirty(id, false, 0) on their next tick.
      if (id === "all") {
        atom = Object.fromEntries(
          SECTION_IDS.map(s => [s, { dirty: false, dirtyFieldCount: 0 }]),
        ) as DirtyMap;
      } else {
        atom = { ...atom, [id]: { dirty: false, dirtyFieldCount: 0 } };
      }
      notify();
    }

    export interface UseDirtySectionsReturn {
      sections: DirtyMap;
      dirtyIds: SectionId[];
      anyDirty: boolean;
    }

    export function useDirtySections(): UseDirtySectionsReturn {
      const map = useSyncExternalStore(subscribe, get, get);
      const dirtyIds = SECTION_IDS.filter(id => map[id].dirty);
      return { sections: map, dirtyIds, anyDirty: dirtyIds.length > 0 };
    }
    ```

     * use-dirty-sections via a single useEffect (no dual-source-of-truth).

      form: UseFormReturn<TValues>,

      const [sectionBannerError, setSectionBannerError] = useState<string | null>(null);

      // can read getDirtySections() without a hook context.

      // and call form.reset() when targeted (or when "all" is emitted).

          if (id === "all" || id === sectionId) {
            form.reset();
          }

              },
            });

            toast.success("Saved.", { duration: 3000 });

          setTimeout(() => setState("idle"), 2000);
        } catch (e) {

          if (mapped) {

            setSectionBannerError(mapped.sectionBannerError);

              },
            });
          } else {
            toast.error(err.message ?? "Save failed.");
          }
          setState("error");
          setTimeout(() => setState("idle"), 2000);
        }
      }, [base, sectionId, refetch, addFields, actions, setActive]);

      return { state, save, fieldErrors, sectionBannerError };
    }
