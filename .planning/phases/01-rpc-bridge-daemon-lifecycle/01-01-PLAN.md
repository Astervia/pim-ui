**W1 checker fix â€” single-listener design:** `subscribeDaemon` ONLY invokes the Rust `daemon_subscribe` command and returns the subscription_id. It does NOT register a Tauri `listen(rpcEvent)` â€” the `useDaemonState` hook (Plan 03) owns exactly one global `listen(DaemonEvents.rpcEvent)` and fans out to registered handlers via an internal per-event Map. ```typescript
/**
 * Typed bridge from the React frontend to the Tauri Rust shell.
 *
 * The Rust side (src-tauri/src/rpc/*) owns the long-lived JSON-RPC
 * connection to pim-daemon over a Unix socket. The frontend talks to
 * the Rust side via Tauri `invoke` commands (request/response) and
 * Tauri event listeners (daemon â†’ UI push).
 *
 * Contract: mirrors proximity-internet-mesh/docs/RPC.md v1.
 * When docs/RPC.md changes, regenerate types in `rpc-types.ts` then
 * adjust Rust command names to match.
 *
 * W1 listener-dedup design: This module does NOT own a Tauri
 * `listen(daemon://rpc-event)` call. subscribeDaemon() only tells Rust
 * to start forwarding the stream and hands back the subscription_id.
 * The ONE global listener lives in src/hooks/use-daemon-state.ts and
 * fans out to per-event subscribers via an internal Map<eventName, Set<handler>>.
 * This keeps exactly one `listen(rpcEvent)` call in the whole app regardless
 * of how many UI components subscribe.
 */

import { invoke } from "@tauri-apps/api/core";
import type {
  RpcMethodMap, RpcMethodName, RpcEventName, RpcError,
} from "./rpc-types";

// Tauri command names â€” EXACT strings the Rust #[tauri::command] handlers
// must use. Kept in sync with src-tauri/src/rpc/commands.rs (Plan 02).
const CMD = {
  call: "daemon_call",           // generic JSON-RPC request/response
  subscribe: "daemon_subscribe", // starts forwarding a tauri event stream
  unsubscribe: "daemon_unsubscribe",
  startDaemon: "daemon_start",   // spawns sidecar (Plan 02)
  stopDaemon: "daemon_stop",     // kills sidecar (Plan 02)
  getLastError: "daemon_last_error",
} as const;

// Tauri event names the Rust side emits via `app.emit(...)`. Match strings
// in src-tauri/src/daemon/events.rs (Plan 02).
const EVT = {
  stateChanged: "daemon://state-changed",  // payload: DaemonStateChange (Plan 03 type)
  rpcEvent: "daemon://rpc-event",           // payload: { event: RpcEventName; params: unknown }
} as const;

/** Typed request â€” method name picks the correct params + result types. */
export async function callDaemon<M extends RpcMethodName>(
  method: M,
  params: RpcMethodMap[M]["params"],
): Promise<RpcMethodMap[M]["result"]> {
  return invoke(CMD.call, { method, params });
}

export interface DaemonSubscription {
  /** subscription_id returned by Rust; pass back to unsubscribe. */
  id: string;
  /** Convenience unsubscribe â€” calls the Rust daemon_unsubscribe command by id. */
  unsubscribe: () => Promise<void>;
}

/**
 * Ask Rust to start forwarding one of the 3 event streams.
 *
 * W1: this function does NOT register a Tauri listener. The calling
 * hook (useDaemonState in Plan 03) owns the single global listener and
 * dispatches fan-out to subscribers by event name. Returning just the
 * subscription_id keeps `grep -c "listen(DaemonEvents.rpcEvent" src/`
 * at exactly 1.
 */
export async function subscribeDaemon(
  event: RpcEventName,
): Promise<DaemonSubscription> {
  // Rust returns a subscription_id (uuid). daemon_unsubscribe accepts the
  // same id and looks up the event name via its internal HashMap.
  const id = await invoke<string>(CMD.subscribe, { event });
  return {
    id,
    unsubscribe: async () => {
