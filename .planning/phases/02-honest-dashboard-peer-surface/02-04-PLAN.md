      return (
        <div className="grid grid-cols-[12ch_1fr] gap-x-3 font-code text-sm leading-[1.7]">
          <span className="text-muted-foreground">{label}</span>
          <span
            className={cn(valueClassName, copyable && "cursor-pointer hover:text-primary")}
            onClick={copyable ? handleCopy : undefined}
            title={copyable ? "click to copy" : undefined}
          >
            {value}
          </span>
        </div>
      );
    }

    **C. `src/hooks/use-peer-troubleshoot-log.ts`:**

    Per-peer ring buffer of last 25 `peers.event` entries. Implementation:

    ```typescript
    interface LogEntry {
      at: string;
      kind: PeerEventKind;
      reason?: string;
    }

    // Module-level state: Map<nodeId, LogEntry[]>
    const buffers = new Map<string, LogEntry[]>();
    const listeners = new Set<() => void>();
    let subscriptionRef: DaemonSubscription | null = null;
    let refCount = 0;

    function pushEntry(nodeId: string, entry: LogEntry) {
      const buf = buffers.get(nodeId) ?? [];
      const next = [entry, ...buf].slice(0, 25);
      buffers.set(nodeId, next);
      listeners.forEach(fn => fn());
    }

    export function usePeerTroubleshootLog(nodeId: string | null | undefined): LogEntry[] {
      const { actions } = useDaemonState();

      useEffect(() => {
        refCount++;
        if (refCount === 1) {
          // First consumer — subscribe to peers.event
          actions.subscribe("peers.event", (evt) => {
            // PeerEvent.peer can be PeerSummary or PeerDiscovered — narrow by presence of node_id field
            const peer = evt.peer as PeerSummary;
            const id = peer.node_id;
            if (!id) return;               // PeerDiscovered with null node_id — skip
            pushEntry(id, { at: evt.at, kind: evt.kind, reason: evt.reason });
          }).then(sub => { subscriptionRef = sub; });
        }
        return () => {
          refCount--;
          if (refCount === 0 && subscriptionRef) {
            subscriptionRef.unsubscribe();
            subscriptionRef = null;
          }
        };
      }, [actions]);

      const snapshot = useSyncExternalStore(
        (cb) => { listeners.add(cb); return () => { listeners.delete(cb); }; },
        () => nodeId ? (buffers.get(nodeId) ?? []) : [],
        () => nodeId ? (buffers.get(nodeId) ?? []) : [],
      );
      return snapshot;
    }
    ```

    Important: this hook registers a SECOND `actions.subscribe("peers.event", ...)` handler alongside the one Plan 02-01 registered for the live peer list. The W1 fan-out supports multiple handlers per event (the `eventHandlers` map holds a Set). Rust-side subscription is idempotent: the hook calls `subscribeDaemon("peers.event")` which may return the SAME subscription_id as the existing one (implementation-defined in Plan 01-02 Rust) — this is fine; both handlers fire on every event, the buffer only pushes for nodes matching the displayed peer's id when `usePeerTroubleshootLog(nodeId)` is called.

    **A. `src/hooks/use-pair-approval.ts`:**

    Manages the queue of pair triggers (inbound + outbound) and current-open state.

    ```typescript
    import { useEffect } from "react";
    import { useSyncExternalStore } from "react";
    import { useDaemonState } from "./use-daemon-state";
    import type { PeerDiscovered, PeerEvent } from "@/lib/rpc-types";

    export interface PairApprovalTrigger {
      mode: "inbound" | "outbound";
      /** Discovered peer data; inbound mode uses what arrived on peers.event */
      discovered: PeerDiscovered;
    }

    let current: PairApprovalTrigger | null = null;
    const queue: PairApprovalTrigger[] = [];
    const listeners = new Set<() => void>();
    let subscribed = false;

    function notify() { listeners.forEach(fn => fn()); }

    function openNext() {
      if (current !== null) return;            // already open
      const next = queue.shift();
      if (next) { current = next; notify(); }
    }

    export function usePairApproval() {
      const { actions } = useDaemonState();

      useEffect(() => {
        if (subscribed) return;
        subscribed = true;
        // Listen for inbound pair-handshake triggers on peers.event { kind: discovered }
        // D-21: daemon emits discovered events; treat all of them as potential pair triggers
        // The inbound filter here is: kind === "discovered" AND peer is a PeerDiscovered shape
        actions.subscribe("peers.event", (evt: PeerEvent) => {
          if (evt.kind !== "discovered") return;
          const discovered = evt.peer as PeerDiscovered;
          // Anonymous peers with null node_id: D-20 says pairing is hidden for those; don't queue
          if (discovered.node_id === null) return;
          queue.push({ mode: "inbound", discovered });
          openNext();
        }).catch((e) => console.warn("pair-approval subscribe failed:", e));
      }, [actions]);

      return {
        current: useSyncExternalStore(cb => (listeners.add(cb), () => listeners.delete(cb)), () => current, () => current),
        queue: queue.length,
        requestOutbound: (d: PeerDiscovered) => {
          queue.push({ mode: "outbound", discovered: d });
          openNext();
        },
        close: () => {
          current = null;
          notify();
          // D-22: if queued, open the next one
          setTimeout(openNext, 0);
        },
      };
    }
    ```

    D-22 queue behavior:
    - First event → opens modal.
    - Second event while modal open → `queue.push(...)`, modal stays on current.
    - User closes current → `setTimeout(openNext, 0)` opens next.
    - Queue depth is internally tracked; we log `console.info("pair queue: N waiting")` for traceability but do NOT surface in UI (D-22).
