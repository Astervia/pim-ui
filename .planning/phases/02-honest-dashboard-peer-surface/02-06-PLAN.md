    **A. `src/components/brand/subscription-error-toast.tsx`:**

    ```tsx
    import { useEffect, useRef } from "react";
    import { toast } from "sonner";
    import { useDaemonState } from "@/hooks/use-daemon-state";

    export function SubscriptionErrorToast() {
      const { snapshot } = useDaemonState();
      const lastShownRef = useRef<string | null>(null);

      useEffect(() => {
        const err = snapshot.subscriptionError;
        if (!err) { lastShownRef.current = null; return; }
        // Dedupe — only fire once per stream until cleared
        const key = `${err.stream}:${err.error.code}`;
        if (lastShownRef.current === key) return;
        lastShownRef.current = key;

        toast.error(
          // D-31 copy VERBATIM:
          `Couldn't subscribe to ${err.stream}. Check the Logs tab.`,
          {
            duration: 6000,
            className: "font-mono",    // brand typography override
            // sonner auto-dismisses on click by default; no action prop needed
          },
        );
      }, [snapshot.subscriptionError]);

      return null;
    }
    ```

    Copy rule VERBATIM (D-31 + UI-SPEC §Subscription-failure toast): `Couldn't subscribe to {stream}. Check the Logs tab.` where `{stream}` is the RPC event name (`"status.event"` / `"peers.event"` / `"logs.event"`). NO exclamation marks.
