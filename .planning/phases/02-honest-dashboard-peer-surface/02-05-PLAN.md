    ```typescript
    import { useCallback, useEffect, useRef, useState, useSyncExternalStore } from "react";
    import { useDaemonState } from "./use-daemon-state";
    import { callDaemon } from "@/lib/rpc";
    import type { LogEvent, LogLevel, DaemonSubscription } from "@/lib/rpc-types";
    // NOTE: DaemonSubscription is declared in src/lib/rpc.ts not rpc-types.ts — correct the import if needed

    export type StreamStatus = "streaming" | "paused" | "reconnecting" | "idle";

    interface UseLogsStreamResult {
      events: LogEvent[];              // visible slice after level+peer filter
      allEvents: LogEvent[];            // full ring buffer (capped at 2000)
      level: LogLevel;
      setLevel: (l: LogLevel) => void;  // triggers server-side re-subscribe
      peerFilter: string | null;        // node_id or null for (all)
      setPeerFilter: (p: string | null) => void;
      status: StreamStatus;
      errorMessage: string | null;
    }

    const MAX_ENTRIES = 2000;

    export function useLogsStream(): UseLogsStreamResult {
      const { actions, snapshot } = useDaemonState();
      const [level, setLevelState] = useState<LogLevel>("info");
      const [peerFilter, setPeerFilter] = useState<string | null>(null);

      // Mutable ring buffer — avoids React state churn on high-volume events.
      // We snapshot-copy when a component re-renders.
      const bufferRef = useRef<LogEvent[]>([]);
      const [, forceRender] = useState(0);
      const bumpRender = useCallback(() => forceRender((n) => n + 1), []);

      const subscriptionIdRef = useRef<string | null>(null);
      const fanOutSubRef = useRef<DaemonSubscription | null>(null);
      const [status, setStatus] = useState<StreamStatus>("idle");
      const [errorMessage, setErrorMessage] = useState<string | null>(null);

      async function subscribeWithRetry(lvl: LogLevel) {
        setStatus("reconnecting");
        // Tear down existing subscription first (level-change path)
        if (subscriptionIdRef.current) {
          try { await callDaemon("logs.unsubscribe", { subscription_id: subscriptionIdRef.current }); }
          catch {}
          subscriptionIdRef.current = null;
        }
        // Try + retry-once (D-31)
        const attempt = async (): Promise<void> => {
          const res = await callDaemon("logs.subscribe", { min_level: lvl, sources: [] });
          subscriptionIdRef.current = res.subscription_id;
        };
        try {
          await attempt();
          setStatus("streaming");
          setErrorMessage(null);
        } catch (e1) {
          await new Promise((r) => setTimeout(r, 500));
          try {
            await attempt();
            setStatus("streaming");
            setErrorMessage(null);
          } catch (e2) {
            setStatus("idle");
            setErrorMessage(e2 instanceof Error ? e2.message : "Subscription failed");
            return;
          }
        }
      }

      // Mount: subscribe + register fan-out handler
      useEffect(() => {
        let alive = true;
        let localSub: DaemonSubscription | null = null;

        (async () => {
          if (!alive) return;
          await subscribeWithRetry(level);
          if (!alive) return;
          localSub = await actions.subscribe("logs.event", (evt) => {
            // D-25 ring buffer: prepend, cap at MAX_ENTRIES, drop oldest (we keep newest-first order
            // for easier slicing; LogList reverses when rendering bottom-to-top)
            const buf = bufferRef.current;
            buf.unshift(evt);
            if (buf.length > MAX_ENTRIES) buf.length = MAX_ENTRIES;
            bumpRender();
          });
          fanOutSubRef.current = localSub;
        })().catch((e) => {
          console.warn("useLogsStream mount:", e);
          setStatus("idle");
          setErrorMessage(e instanceof Error ? e.message : "Subscription failed");
        });

        return () => {
          alive = false;
          // D-25 unmount: unsubscribe fan-out handler + tell daemon to stop forwarding
          if (localSub) localSub.unsubscribe().catch(() => {});
          if (subscriptionIdRef.current) {
            callDaemon("logs.unsubscribe", { subscription_id: subscriptionIdRef.current }).catch(() => {});
            subscriptionIdRef.current = null;
          }
        };
        // eslint-disable-next-line react-hooks/exhaustive-deps
      }, []);

      // When level changes, re-subscribe server-side (D-26)
      const setLevel = useCallback((lvl: LogLevel) => {
        setLevelState(lvl);
        void subscribeWithRetry(lvl);
      }, []);

      // Client-side peer filter (D-26). Level is already server-side.
      const filteredEvents: LogEvent[] = peerFilter
        ? bufferRef.current.filter((e) => e.peer_id === peerFilter)
        : bufferRef.current;

      return {
        events: filteredEvents,
        allEvents: bufferRef.current,
        level,
        setLevel,
        peerFilter,
        setPeerFilter,
        status,
        errorMessage,
      };
    }
    ```

    Rules:
    - Ring buffer capped at 2000 (D-25).
    - Newest-first order in the buffer; `LogList` renders reverse so oldest → newest top-to-bottom (terminal convention) OR keep newest-first with reverse-scroll. Decision: buffer holds newest-first; `LogList` renders `allEvents[...].slice().reverse()` so the DOM order is oldest-at-top / newest-at-bottom, which matches UI-SPEC §S5 ASCII (21:14:07 first, 21:14:09 last).
    - Level filter → server-side re-subscribe (D-26). The old `subscription_id` is unsubscribed before the new one is acquired.
    - Peer filter → client-side `filter()` on buffer (D-26 — daemon's `sources` is module-based not peer-based).
    - On mount: retry-once per D-31.
    - On level change: retry-once per D-31.
    - On unmount: unsubscribe both the fan-out handler AND the daemon subscription.

    No component renders this hook in this task — Task 3 and Task 4 do.

    ```tsx
    import { useEffect, useMemo, useRef, useState } from "react";
    import { FixedSizeList, type ListChildComponentProps } from "react-window";
    import type { LogEvent } from "@/lib/rpc-types";
    import { LogRow } from "./log-row";
    import { cn } from "@/lib/utils";

    const ROW_HEIGHT = 22;   // font-code text-sm leading-[1.5] + py-0.5 ≈ 22px
    const STICKY_THRESHOLD_PX = 40;   // D-28

    interface LogListProps { events: LogEvent[]; }   // newest-first from hook — we reverse below

    export function LogList({ events }: LogListProps) {
      // Reverse so oldest-top / newest-bottom (terminal convention)
      const display = useMemo(() => [...events].reverse(), [events]);
      const listRef = useRef<FixedSizeList>(null);
      const [isSticky, setIsSticky] = useState(true);
      const [newCount, setNewCount] = useState(0);
      const prevLength = useRef(display.length);

      useEffect(() => {
        const delta = display.length - prevLength.current;
        prevLength.current = display.length;
        if (delta > 0) {
          if (isSticky && listRef.current) {
            listRef.current.scrollToItem(display.length - 1, "end");
          } else {
            setNewCount((n) => n + delta);
          }
        }
      }, [display.length, isSticky]);

      function onScroll({ scrollOffset, scrollUpdateWasRequested }: { scrollOffset: number; scrollUpdateWasRequested: boolean }) {
        if (scrollUpdateWasRequested) return;
        const height = 400;                // match FixedSizeList height below
        const contentHeight = display.length * ROW_HEIGHT;
        const atBottom = scrollOffset >= contentHeight - height - STICKY_THRESHOLD_PX;
        setIsSticky(atBottom);
        if (atBottom) setNewCount(0);
      }

      function jumpToBottom() {
        setIsSticky(true);
        setNewCount(0);
        listRef.current?.scrollToItem(display.length - 1, "end");
      }

      const Row = ({ index, style }: ListChildComponentProps) => (
        <div style={style}><LogRow event={display[index]} /></div>
      );

      return (
        <div className="relative" role="log" aria-live={isSticky ? "polite" : "off"}>
          <FixedSizeList
            ref={listRef}
            height={400}
            itemCount={display.length}
            itemSize={ROW_HEIGHT}
            width="100%"
            onScroll={onScroll}
          >
            {Row}
          </FixedSizeList>

          {(!isSticky || newCount > 0) && (
            <button
              type="button"
              onClick={jumpToBottom}
              className={cn(
                "absolute bottom-4 right-4",
                "bg-primary text-primary-foreground border border-primary",
                "hover:bg-background hover:text-primary",
                "px-3 py-1 font-mono text-xs uppercase tracking-wider",
                "focus-visible:outline focus-visible:outline-1 focus-visible:outline-ring focus-visible:outline-offset-2",
              )}
            >
              [ {newCount > 0 ? `${newCount} new` : "paused"} · jump to bottom ]
            </button>
          )}
        </div>
      );
    }
    ```

    Pill copy (UI-SPEC §"Jump to bottom" pill):
    - When `newCount > 0`: `[ {N} new · jump to bottom ]`
    - When sticky paused but no new: `[ paused · jump to bottom ]`

    Auto-scroll threshold: 40px (D-28 + UI-SPEC).
    Row height: 22px (font-code text-sm leading-[1.5] renders approximately at this density — if execution shows otherwise, adjust based on measured metrics; don't change the UI-SPEC).
