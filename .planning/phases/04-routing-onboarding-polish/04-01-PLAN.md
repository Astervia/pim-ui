    export interface PreflightRow {

    }

    export interface PreflightResult {
      ok: boolean;          // every row passing
      rows: PreflightRow[]; // exactly 3, in fixed order: interface, gateway, split-default
    }

    /** D-14: pure assembly of the runtime line. Returns null when route_on=false or status=null. */
    export function formatRouteLine(
      status: Status | null,
      routeTable: RouteTableResult | null,
    ): string | null {
      if (status === null) return null;
      if (status.route_on === false) return null;
      const sel = status.routes.selected_gateway;
      if (sel === null) return null;
      const gwPeer = status.peers.find((p) => p.node_id === sel) ?? null;
      const gatewayLabel = gwPeer !== null && gwPeer.label !== null
        ? gwPeer.label
        : `gateway-${sel.slice(0, 8)}`;

      // First-hop derivation: find route entry whose destination is "internet" or whose via is selected_gateway.
      let firstHopId: string | null = null;
      if (routeTable !== null) {
        const internetRoute = routeTable.routes.find(
          (r) => r.destination === "internet" || r.is_gateway === true,
        );
        if (internetRoute !== undefined) {
          firstHopId = internetRoute.via;
        }
      }

      // Direct (single-hop) gateway: via === selected_gateway, or peer.route_hops === 1, or no route table info → render no parenthetical.
      if (firstHopId === null || firstHopId === sel) {
        if (gwPeer !== null && gwPeer.route_hops <= 1) {
          return `Routing through ${gatewayLabel}`;
        }
        if (firstHopId === null) {
          return `Routing through ${gatewayLabel}`;
        }
      }
      // Multi-hop: lookup first-hop peer and render relay label.
      const hopPeer = status.peers.find((p) => p.node_id === firstHopId) ?? null;
      const hopLabel = hopPeer !== null && hopPeer.label !== null
        ? hopPeer.label
        : firstHopId !== null
          ? `relay-${firstHopId.slice(0, 8)}`
          : "relay-?";
      return `Routing through ${gatewayLabel} (via ${hopLabel})`;
    }

    /** D-12: pure derivation of the three pre-flight rows. */
    export function derivePreflight(status: Status | null): PreflightResult {
      if (status === null) return { ok: false, rows: [] };
      const ifaceOk = status.interface.up === true;
      const ifaceMsg = ifaceOk
        ? PREFLIGHT_INTERFACE_UP_TEMPLATE.replace("{iface_name}", status.interface.name)
        : PREFLIGHT_INTERFACE_DOWN_TEMPLATE.replace("{iface_name}", status.interface.name);

      const sel = status.routes.selected_gateway;
      const gwPeer: PeerSummary | null = sel === null
        ? null
        : status.peers.find((p) => p.node_id === sel) ?? null;
      const gatewayOk = sel !== null
        && gwPeer !== null
        && (gwPeer.state === "active" || gwPeer.state === "relayed");
      const gatewayMsg = gatewayOk && gwPeer !== null
        ? PREFLIGHT_GATEWAY_REACHABLE_TEMPLATE
            .replace("{label}", gwPeer.label === null ? `gateway-${sel!.slice(0, 8)}` : gwPeer.label)
            .replace("{latency}", String(gwPeer.latency_ms === null ? "?" : gwPeer.latency_ms))
        : PREFLIGHT_NO_GATEWAY;

      // Split-default support: compile-time guarantee in v1 (route.set_split_default exists in RpcMethodMap). Always pass.
      const splitOk = true;
      const splitMsg = PREFLIGHT_SPLIT_DEFAULT_SUPPORTED;

      const rows: PreflightRow[] = [
        { ok: ifaceOk, msg: ifaceMsg },
        { ok: gatewayOk, msg: gatewayMsg },
        { ok: splitOk, msg: splitMsg },
      ];
      return { ok: rows.every((r) => r.ok === true), rows };
    }

    import { readFile, readdir, stat } from "node:fs/promises";
    import { join } from "node:path";

    const BANNED_PHRASES = [
      "Add your first peer",
      "Welcome to pim",
      "Get started",
      "Connecting…",  // note: uses U+2026 ellipsis; D-26 §3 verbatim

      "Whoops",

    // Parse §3 and §4 — fall back to the hardcoded list on parse failure.
    async function loadFromCopyDoc() {
      try {
        const md = await readFile("docs/COPY.md", "utf8");
        // Section 3 lines: "- " + literal phrase
        // Section 4 lines: same
        const sec3 = md.match(/## 3\. Banned phrases\n([\s\S]*?)(?=\n## )/);
        const sec4 = md.match(/## 4\. Soft warnings.*?\n([\s\S]*?)(?=\n## )/);
        if (sec3 !== null) {
          const phrases = [...sec3[1].matchAll(/^- (.+)$/gm)].map((m) => m[1].trim()).filter((s) => s.length > 0 && s.startsWith("any string") === false);
          if (phrases.length > 0) BANNED_PHRASES.length = 0, BANNED_PHRASES.push(...phrases);
        }

    async function* walk(dir) {
      for (const e of await readdir(dir, { withFileTypes: true })) {
        const p = join(dir, e.name);
        if (e.isDirectory() === true) yield* walk(p);
        else if ((p.endsWith(".ts") === true || p.endsWith(".tsx") === true) && p.endsWith(".test.ts") === false) yield p;

    function scan(file, text) {
      const violations = [];
      const warnings = [];
      // Strip block comments to reduce false positives.
      const stripped = text.replace(/\/\*[\s\S]*?\*\//g, "");
      const lines = stripped.split("\n");
      for (let i = 0; i < lines.length; i++) {
        const line = lines[i];
        const trimmed = line.trim();
        if (trimmed.startsWith("//") === true) continue;
        if (trimmed.startsWith("import ") === true) continue;

        // Heuristic: '!' inside a string literal between quotes counts as a violation.
        // Match: "...!..." or '...!...' or `...!...` (and JSX-text >...!...<)

          // Skip operator forms: "!==" etc are handled by the line check below.
          if (m[2].includes("!=") === true) continue;
          violations.push(`${file}:${i + 1}: '!' in string literal: ${m[0]}`);

        // JSX text region: between > and < on the same line, exclude operator combinations.
        const jsxMatches = [...line.matchAll(/>([^<>{}\n]*?!.+?)</g)];
        for (const m of jsxMatches) violations.push(`${file}:${i + 1}: '!' in JSX text: ${m[1]}`);

        for (const banned of BANNED_PHRASES) {
          if (line.includes(banned) === true) violations.push(`${file}:${i + 1}: banned phrase '${banned}'`);
        }
        for (const soft of SOFT_WARNINGS) {
          // Only warn when soft word is inside a quoted string literal.
          const re = new RegExp(`(["'\`])[^"'\`\\n]*\\b${soft}\\b[^"'\`\\n]*\\1`, "i");
          if (re.test(line) === true) warnings.push(`${file}:${i + 1}: soft warning '${soft}': ${line.trim()}`);
        }
      }
      return { violations, warnings };
    }

        process.exit(1);
      }
      console.log(`audit:copy: 0 hard violations, ${allWarnings.length} soft warnings`);
    }

    main().catch((e) => { console.log(`audit:copy crashed: ${e.message}`); process.exit(1); });
