    ```typescript
    /**
     * Phase 4 D-18: useRouteTable — one-shot fetch + status.event-driven refetch.
     *
     * W1 contract: this hook DOES NOT register a Tauri event listener. It
     * joins the existing fan-out via `useDaemonState().actions.subscribe`,
     * which the W1 owner (use-daemon-state.ts) routes through its single
     * `eventHandlers` map.
     *
     * Refetch trigger kinds (D-19): route_on, route_off, gateway_selected,
     * gateway_lost, kill_switch. peers.event is intentionally excluded;
     * peer flap does not change the routing table from the daemon's POV.
     *
     * D-20 escape hatch: refetch() is exported so a consumer can render a
     * `[ refresh ]` button when the daemon misses an event.
     */
    import { useEffect, useState, useRef, useCallback } from "react";
    import { useDaemonState } from "./use-daemon-state";
    import { callDaemon } from "@/lib/rpc";
    import type { RouteTableResult, RpcError, StatusEvent, StatusEventKind } from "@/lib/rpc-types";

    export interface UseRouteTableResult {
      table: RouteTableResult | null;
      loading: boolean;
      error: RpcError | null;
      refetch: () => Promise<void>;
    }

    // Module-level refcount + shared state (mirrors usePeerTroubleshootLog pattern).
    let refcount = 0;
    let unsubscribe: (() => Promise<void>) | null = null;
    let sharedTable: RouteTableResult | null = null;
    let sharedError: RpcError | null = null;
    let sharedLoading = false;
    const subscribers = new Set<() => void>();

    function notifyAll() { subscribers.forEach((fn) => fn()); }

    const REFETCH_KINDS: ReadonlySet<StatusEventKind> = new Set([
      "route_on",
      "route_off",
      "gateway_selected",
      "gateway_lost",
      "kill_switch",
    ]);

    async function doFetch(): Promise<void> {
      sharedLoading = true; sharedError = null; notifyAll();
      try {
        const t = await callDaemon("route.table", null);
        sharedTable = t; sharedLoading = false; notifyAll();
      } catch (e) {
        sharedError = e && typeof e === "object" && "code" in e
          ? (e as RpcError)
          : { code: -32603, message: e instanceof Error ? e.message : String(e) };
        sharedLoading = false; notifyAll();
      }
    }

    export function useRouteTable(): UseRouteTableResult {
      const { actions } = useDaemonState();
      const [, force] = useState(0);
      const tickRef = useRef(0);

      useEffect(() => {
        const sub = () => { tickRef.current++; force((n) => n + 1); };
        subscribers.add(sub);
        refcount++;
        if (refcount === 1) {
          // First consumer — fetch and subscribe.
          doFetch();
          actions.subscribe("status.event", (evt: StatusEvent) => {
            if (REFETCH_KINDS.has(evt.kind) === true) doFetch();
          }).then((s) => { unsubscribe = s.unsubscribe; });
        }
        return () => {
          subscribers.delete(sub);
          refcount--;
          if (refcount === 0 && unsubscribe !== null) {
            const u = unsubscribe; unsubscribe = null;
            u().catch(() => {});
            sharedTable = null; sharedError = null; sharedLoading = false;
          }
        };
      }, [actions]);

      const refetch = useCallback(() => doFetch(), []);

      return { table: sharedTable, loading: sharedLoading, error: sharedError, refetch };
    }

    // Exposed for tests.
    export const __test_resetRouteTable = (): void => {
      refcount = 0; unsubscribe = null;
      sharedTable = null; sharedError = null; sharedLoading = false;
      subscribers.clear();
    };
    ```
